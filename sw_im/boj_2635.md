```python
n = int(input())
max = 0
max_array = []
for i in range(n//2+1, n+1):
    array = [n, i]
    while True:
        if array[-2] - array[-1] >= 0:
            array.append(array[-2]-array[-1])
        else:
            break
    if len(array) > max:
        max = len(array)
        max_array = array
print(max)
for i in max_array:
    print(i, end=" ")
```
