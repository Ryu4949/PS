for _ in range(4):
    data = list(map(int, input().split()))
    sq1_width = data[2]-data[0]
    sq1_height = data[3]-data[1]
    sq2_width = data[6]-data[4]
    sq2_height = data[7]-data[5]

    total_width = max(data[2], data[6]) - min(data[0], data[4])
    total_height = max(data[3], data[7]) - min(data[1], data[5])

    if sq1_width + sq2_width > total_width:
        if sq1_height + sq2_height > total_height:
            print("a")
        elif sq1_height + sq2_height == total_height:
            print("b")
        else:
            print("d")
    elif sq1_width + sq2_width == total_width:
        if sq1_height + sq2_height > total_height:
            print("b")
        elif sq1_height + sq2_height == total_height:
            print("c")
        else:
            print("d")
    else:
        print("d")
