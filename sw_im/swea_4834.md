t = int(input())
for num in range(1, t+1):
    n = int(input())
    card = list(map(int, list(input())))
    #카드의 장수를 담아줄 리스트
    cnt = [0] * 10
    
    #최대 장수인 카드가 여러 종류일 경우 가장 큰 카드번호를 출력하기 위해
    #9 -> 0 순서로 장수를 담음
    for i in card:
        cnt[9-i] += 1
    print(f'#{num} {9-cnt.index(max(cnt))} {max(cnt)}')
