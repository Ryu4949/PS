t = int(input())
for num in range(1, t+1):
    n = int(input())
    nums = list(map(int, input().split()))
    nums.sort()

    #오름차순 정렬해주고 끝에서 하나 앞에서 하나씩 총 5번 출력
    print(f'#{num}', end=' ')
    for i in range(5):
        print(nums[-1-i], end=' ')
        print(nums[i], end=' ')
    print()
