n = int(input())
lst = list(map(int, input().split()))
count_inc = ''
count_dec = ''
for i in range(n):
    if i == 0:
        count_inc += '+'
        count_dec += '-'
    else:
        if lst[i] > lst[i-1]:
            count_inc += '+'
            count_dec += '+-'
        elif lst[i] < lst[i-1]:
            count_inc += '-+'
            count_dec += '-'
        else:
            count_inc += '+'
            count_dec += '-'

inc = max(list(count_inc.split('-')), key=lambda x:len(x))
dec = max(list(count_dec.split('+')), key=lambda x:len(x))
print(max(len(inc), len(dec)))
